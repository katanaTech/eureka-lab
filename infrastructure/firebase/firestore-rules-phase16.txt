// Phase 16 — Inventory & Tenant rules
// To be merged into the main firestore.rules file

// Inventories — user can read/write own; admin SDK for atomic purchases
match /inventories/{userId} {
  allow read: if request.auth != null && request.auth.uid == userId;
  allow write: if request.auth != null && request.auth.uid == userId;

  match /dailyEarn/{date} {
    allow read: if request.auth != null && request.auth.uid == userId;
    allow write: if request.auth != null && request.auth.uid == userId;
  }
}

// Tenants — only admins/teachers can read/write
match /tenants/{tenantId} {
  allow read: if request.auth != null &&
    (request.auth.token.role == 'admin' || request.auth.token.role == 'teacher');
  allow write: if request.auth != null && request.auth.token.role == 'admin';
}
